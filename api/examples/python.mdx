    ```python
    import requests
    import base64
    from typing import Optional, Dict, List

    class PrintBridgeClient:
        def __init__(self, base_url: str, api_token: str):
            self.base_url = base_url.rstrip('/')
            self.api_token = api_token
            self.session = requests.Session()
            self.session.headers.update({'x-api-token': api_token})
        
        def _request(self, method: str, endpoint: str, **kwargs) -> Dict:
            """Make HTTP request to the API."""
            url = f"{self.base_url}{endpoint}"
            
            try:
                response = self.session.request(method, url, **kwargs)
                response.raise_for_status()
                return response.json()
            except requests.exceptions.HTTPError as e:
                error_msg = e.response.json().get('error', str(e))
                raise Exception(f"API Error: {error_msg}")
            except Exception as e:
                raise Exception(f"Request failed: {str(e)}")
        
        def check_health(self) -> Dict:
            """Check if the service is running."""
            response = requests.get(f"{self.base_url}/health")
            return response.json()
        
        def get_printers(self) -> List[Dict]:
            """Get list of available printers."""
            return self._request('GET', '/api/printers')
        
        def print_document(
            self,
            content: str,
            content_type: str,
            printer_name: Optional[str] = None,
            copies: Optional[int] = None,
            options: Optional[Dict] = None
        ) -> Dict:
            """Print a document.
            
            Args:
                content: Document content (base64 for files, plain text for text)
                content_type: Type of content ('pdf', 'text', 'html', 'image')
                printer_name: Name of printer (optional, uses default if not specified)
                copies: Number of copies (optional, default: 1)
                options: Print options dict (optional)
            
            Returns:
                Dict with success status, message, and job_id
            """
            payload = {
                'content': content,
                'content_type': content_type
            }
            
            if printer_name:
                payload['printer_name'] = printer_name
            if copies:
                payload['copies'] = copies
            if options:
                payload['options'] = options
            
            return self._request('POST', '/api/print', json=payload)
        
        @staticmethod
        def encode_file(file_path: str) -> str:
            """Read a file and encode it to base64."""
            with open(file_path, 'rb') as f:
                return base64.b64encode(f.read()).decode('utf-8')
        
        def print_pdf(
            self,
            file_path: str,
            printer_name: Optional[str] = None,
            copies: int = 1
        ) -> Dict:
            """Print a PDF file."""
            content = self.encode_file(file_path)
            return self.print_document(content, 'pdf', printer_name, copies)
        
        def print_text(
            self,
            text: str,
            printer_name: Optional[str] = None,
            copies: int = 1
        ) -> Dict:
            """Print plain text."""
            return self.print_document(text, 'text', printer_name, copies)
        
        def print_html(
            self,
            html: str,
            printer_name: Optional[str] = None,
            copies: int = 1
        ) -> Dict:
            """Print HTML content."""
            return self.print_document(html, 'html', printer_name, copies)
        
        def print_image(
            self,
            file_path: str,
            printer_name: Optional[str] = None,
            copies: int = 1
        ) -> Dict:
            """Print an image file."""
            content = self.encode_file(file_path)
            return self.print_document(content, 'image', printer_name, copies)

    # Initialize client
    client = PrintBridgeClient(
        base_url='http://localhost:8765',
        api_token='your_api_token_here'
    )
    ```

    ```python
    import httpx
    import base64
    from typing import Optional, Dict, List

    class AsyncPrintBridgeClient:
        def __init__(self, base_url: str, api_token: str):
            self.base_url = base_url.rstrip('/')
            self.api_token = api_token
            self.headers = {'x-api-token': api_token}
        
        async def _request(self, method: str, endpoint: str, **kwargs) -> Dict:
            """Make async HTTP request to the API."""
            url = f"{self.base_url}{endpoint}"
            
            # Merge headers
            headers = {**self.headers, **kwargs.pop('headers', {})}
            
            async with httpx.AsyncClient() as client:
                try:
                    response = await client.request(
                        method, url, headers=headers, **kwargs
                    )
                    response.raise_for_status()
                    return response.json()
                except httpx.HTTPStatusError as e:
                    error_msg = e.response.json().get('error', str(e))
                    raise Exception(f"API Error: {error_msg}")
                except Exception as e:
                    raise Exception(f"Request failed: {str(e)}")
        
        async def check_health(self) -> Dict:
            """Check if the service is running."""
            async with httpx.AsyncClient() as client:
                response = await client.get(f"{self.base_url}/health")
                return response.json()
        
        async def get_printers(self) -> List[Dict]:
            """Get list of available printers."""

        async def print_document(
    ```

### List Available Printers

<Tabs>
  <Tab title="Requests">
    ```python
    import requests

    def list_printers():
        headers = {'x-api-token': 'your_api_token_here'}
        response = requests.get(
            'http://localhost:8765/api/printers',
            headers=headers
        )
        response.raise_for_status()
        
        printers = response.json()
        
        print("Available Printers:")
        for printer in printers:
            default = " (Default)" if printer['is_default'] else ""
            color = " [Color]" if printer['supports_color'] else ""
            print(f"  - {printer['name']}{default}{color}")
            print(f"    Status: {printer['status']}")
            print(f"    Paper sizes: {', '.join(printer['paper_sizes'])}")
        
        return printers

    # Run
    printers = list_printers()
    ```
  </Tab>

### Error Handling

```python
from print_bridge_client import PrintBridgeClient
import requests

client = PrintBridgeClient(
    base_url='http://localhost:8765',
    api_token='your_api_token_here'
)

def safe_print(content, content_type, printer_name=None):
    """Print with comprehensive error handling."""
    try:
        result = client.print_document(
            content=content,
            content_type=content_type,
            printer_name=printer_name
        )
        return {'success': True, 'data': result}
    
    except Exception as e:
        error_msg = str(e)
        
        # Handle specific error types
        if 'No autorizado' in error_msg:
            return {
                'success': False,
                'error': 'UNAUTHORIZED',
                'message': 'Invalid API token'
            }
        elif 'Límite de velocidad excedido' in error_msg:
            return {
                'success': False,
                'error': 'RATE_LIMIT',
                'message': 'Too many requests. Please try again later.'
            }
        elif 'Formato no soportado' in error_msg:
            return {
                'success': False,
                'error': 'UNSUPPORTED_FORMAT',
                'message': 'File format not supported'
            }
        elif 'Archivo demasiado grande' in error_msg:
            return {
                'success': False,
                'error': 'FILE_TOO_LARGE',
                'message': 'File exceeds maximum size limit'
            }
        else:
            return {
                'success': False,
                'error': 'UNKNOWN',
                'message': error_msg
            }

# Usage
result = safe_print('Test document', 'text', 'HP LaserJet Pro')
if result['success']:
    print(f"Printed successfully: {result['data']['job_id']}")
else:
    print(f"Print failed ({result['error']}): {result['message']}")
```

### Batch Printing

```python
from print_bridge_client import PrintBridgeClient
import time
from typing import List, Dict

client = PrintBridgeClient(
    base_url='http://localhost:8765',
    api_token='your_api_token_here'
)

def batch_print_pdfs(file_paths: List[str], printer_name: str = None, delay: float = 0.5):
    """Print multiple PDF files with rate limiting."""
    results = []
    
    for i, file_path in enumerate(file_paths, 1):
        try:
            print(f"Printing {i}/{len(file_paths)}: {file_path}")
            result = client.print_pdf(file_path, printer_name)
            results.append({
                'file': file_path,
                'success': True,
                'job_id': result['job_id']
            })
            print(f"  ✓ Job ID: {result['job_id']}")
            
            # Rate limiting delay
            if i < len(file_paths):
                time.sleep(delay)
                
        except Exception as e:
            results.append({
                'file': file_path,
                'success': False,
                'error': str(e)
            })
            print(f"  ✗ Failed: {e}")
    
    # Summary
    successful = sum(1 for r in results if r['success'])
    print(f"\nBatch complete: {successful}/{len(file_paths)} successful")
    
    return results

# Usage
files = ['invoice1.pdf', 'invoice2.pdf', 'invoice3.pdf']
results = batch_print_pdfs(files, 'HP LaserJet Pro')
```

### Async Batch Printing (HTTPX)

```python
import asyncio
from typing import List, Dict
from async_print_bridge_client import AsyncPrintBridgeClient

async_client = AsyncPrintBridgeClient(
    base_url='http://localhost:8765',
    api_token='your_api_token_here'
)

async def batch_print_async(file_paths: List[str], printer_name: str = None):
    """Print multiple files concurrently."""
    tasks = [
        async_client.print_pdf(file_path, printer_name)
        for file_path in file_paths
    ]
    
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # Process results
    for i, (file_path, result) in enumerate(zip(file_paths, results), 1):
        if isinstance(result, Exception):
            print(f"{i}. {file_path}: Failed - {result}")
        else:
            print(f"{i}. {file_path}: Success - Job {result['job_id']}")
    
    return results

# Usage
files = ['doc1.pdf', 'doc2.pdf', 'doc3.pdf']
asyncio.run(batch_print_async(files, 'HP LaserJet Pro'))
```

### Django Integration Example

```python
# views.py
from django.http import JsonResponse
from django.views.decorators.http import require_http_methods
from django.conf import settings
from print_bridge_client import PrintBridgeClient
import base64

client = PrintBridgeClient(
    base_url=settings.PRINT_BRIDGE_URL,
    api_token=settings.PRINT_BRIDGE_TOKEN
)

@require_http_methods(["POST"])
def print_document(request):
    """Django view to handle print requests."""
    try:
        # Get uploaded file
        uploaded_file = request.FILES.get('file')
        if not uploaded_file:
            return JsonResponse({'error': 'No file provided'}, status=400)
        
        # Get parameters
        printer_name = request.POST.get('printer_name')
        copies = int(request.POST.get('copies', 1))
        
        # Determine content type
        content_type = 'pdf' if uploaded_file.name.endswith('.pdf') else 'image'
        
        # Encode file content
        file_content = base64.b64encode(uploaded_file.read()).decode('utf-8')
        
        # Print document
        result = client.print_document(
            content=file_content,
            content_type=content_type,
            printer_name=printer_name,
            copies=copies
        )
        
        return JsonResponse({
            'success': True,
            'job_id': result['job_id'],
            'message': result['message']
        })
        
    except Exception as e:
        return JsonResponse({
            'success': False,
            'error': str(e)
        }, status=500)

@require_http_methods(["GET"])
def list_printers(request):
    """Django view to list available printers."""
    try:
        printers = client.get_printers()
        return JsonResponse({'printers': printers})
    except Exception as e:
        return JsonResponse({'error': str(e)}, status=500)
```

## Best Practices

1. **Environment Variables**: Always use environment variables for API tokens
2. **Error Handling**: Wrap all API calls in try-except blocks
3. **Rate Limiting**: Implement delays between bulk print operations
4. **File Validation**: Validate file size and type before encoding
5. **Connection Pooling**: Reuse session objects for multiple requests
6. **Timeouts**: Set appropriate timeouts for print operations
7. **Logging**: Log print operations for audit trails

