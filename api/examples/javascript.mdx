This guide provides comprehensive JavaScript examples for integrating Print My Bridge into your web applications or Node.js services.

## Prerequisites

Before using these examples:

1. Ensure Print My Bridge is running (default: `http://localhost:8765`)
2. Generate an API token through the application UI
3. Install required dependencies (if using Node.js)

## Setup and Configuration

### Browser (Fetch API)

No installation required - use the native Fetch API:

```javascript
const PRINT_BRIDGE_CONFIG = {
  baseURL: 'http://localhost:8765',
  apiToken: 'your_api_token_here'
};
```

### Node.js (Native Fetch - Node 18+)

```javascript
// No installation needed for Node.js 18+
const PRINT_BRIDGE_CONFIG = {
  baseURL: 'http://localhost:8765',
  apiToken: process.env.PRINT_BRIDGE_TOKEN || 'your_api_token_here'
```

## Helper Class

Create a reusable client class for all API operations:

```javascript
class PrintBridgeClient {
  constructor(baseURL, apiToken) {
    this.baseURL = baseURL;
    this.apiToken = apiToken;
  }

  async request(endpoint, options = {}) {
    const url = `${this.baseURL}${endpoint}`;
    const headers = {
      'Content-Type': 'application/json',
      ...options.headers
    };

    // Add auth header for protected endpoints
    if (endpoint !== '/health') {
      headers['x-api-token'] = this.apiToken;
    }

    const config = {
      ...options,
      headers
    };

    try {
      const response = await fetch(url, config);
      
      if (!response.ok) {
        const error = await response.json().catch(() => ({
          error: `HTTP ${response.status}: ${response.statusText}`
        }));
        throw new Error(error.error || 'Request failed');
      }

      return await response.json();
    } catch (error) {
      console.error(`Print Bridge API Error: ${error.message}`);
      throw error;
    }
  }

  async checkHealth() {
    return this.request('/health');
  }

  async getPrinters() {
    return this.request('/api/printers');
  }

  async print(options) {
    return this.request('/api/print', {
      method: 'POST',
      body: JSON.stringify(options)
    });
  }

  // Helper to convert file to base64
  async fileToBase64(file) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = () => {
        const base64 = reader.result.split(',')[1];
        resolve(base64);
      };
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }

  // Helper for Node.js to read file
  async fileToBase64Node(filePath) {
    const fs = require('fs').promises;
    const buffer = await fs.readFile(filePath);
    return buffer.toString('base64');
  }
}

// Initialize client
const client = new PrintBridgeClient(
  'http://localhost:8765',
  'your_api_token_here'
);
```

### Print with Options

```javascript
async function printWithOptions(content, contentType, options = {}) {
  try {
    const result = await client.print({
      content: content,
      content_type: contentType,
      printer_name: options.printer,
      copies: options.copies || 1,
      options: {
        paper_size: options.paperSize || 'A4',
        orientation: options.orientation || 'portrait',
        color: options.color !== false,
        duplex: options.duplex || false
      }
    });

    return result;
  } catch (error) {
    console.error('Print with options failed:', error.message);
    throw error;
  }
}

// Usage
const fs = require('fs').promises;
const buffer = await fs.readFile('document.pdf');
const base64Content = buffer.toString('base64');

await printWithOptions(base64Content, 'pdf', {
  printer: 'HP LaserJet Pro',
  copies: 2,
  paperSize: 'Letter',
  orientation: 'landscape',
  color: false,
  duplex: true
});
```

### Error Handling

```javascript
async function safePrint(content, contentType, printerName = null) {
  try {
    const result = await client.print({
      content: content,
      content_type: contentType,
      printer_name: printerName
    });
    
    return { success: true, data: result };
  } catch (error) {
    console.error('Print error:', error.message);
    
    // Handle specific errors
    if (error.message.includes('No autorizado')) {
      return { success: false, error: 'UNAUTHORIZED', message: 'Invalid API token' };
    } else if (error.message.includes('Límite de velocidad excedido')) {
      return { success: false, error: 'RATE_LIMIT', message: 'Too many requests. Please try again later.' };
    } else if (error.message.includes('Formato no soportado')) {
      return { success: false, error: 'UNSUPPORTED_FORMAT', message: 'File format not supported' };
    } else if (error.message.includes('Archivo demasiado grande')) {
      return { success: false, error: 'FILE_TOO_LARGE', message: 'File exceeds maximum size limit' };
    } else {
      return { success: false, error: 'UNKNOWN', message: error.message };
    }
  }
}

// Usage
const result = await safePrint('Test document', 'text', 'HP LaserJet Pro');
if (result.success) {
  console.log('Printed successfully:', result.data.job_id);
} else {
  console.error(`Print failed (${result.error}): ${result.message}`);
}
```

### Complete React Component Example

```jsx
import React, { useState, useEffect } from 'react';
import { PrintBridgeClient } from './PrintBridgeClient';

const PrintComponent = () => {
  const [client] = useState(() => new PrintBridgeClient(
    'http://localhost:8765',
    process.env.REACT_APP_PRINT_BRIDGE_TOKEN
  ));
  
  const [printers, setPrinters] = useState([]);
  const [selectedPrinter, setSelectedPrinter] = useState('');
  const [status, setStatus] = useState('');
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    loadPrinters();
  }, []);

  const loadPrinters = async () => {
    try {
      const printerList = await client.getPrinters();
      setPrinters(printerList);
      
      // Select default printer
      const defaultPrinter = printerList.find(p => p.is_default);
      if (defaultPrinter) {
        setSelectedPrinter(defaultPrinter.name);
      }
    } catch (error) {
      setStatus(`Failed to load printers: ${error.message}`);
    }
  };

  const handlePrintFile = async (event) => {
    const file = event.target.files[0];
    if (!file) return;

    setLoading(true);
    setStatus('Printing...');

    try {
      const base64Content = await client.fileToBase64(file);
      const fileType = file.name.endsWith('.pdf') ? 'pdf' : 'image';

      const result = await client.print({
        content: base64Content,
        content_type: fileType,
        printer_name: selectedPrinter
      });

      setStatus(`Print successful! Job ID: ${result.job_id}`);
    } catch (error) {
      setStatus(`Print failed: ${error.message}`);
    } finally {
      setLoading(false);
    }
  };

  const handlePrintText = async () => {
    const text = prompt('Enter text to print:');
    if (!text) return;

    setLoading(true);
    setStatus('Printing...');

    try {
      const result = await client.print({
        content: text,
        content_type: 'text',
        printer_name: selectedPrinter
      });

      setStatus(`Print successful! Job ID: ${result.job_id}`);
    } catch (error) {
      setStatus(`Print failed: ${error.message}`);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div>
      <h2>Print My Bridge</h2>
      
      <div>
        <label>Select Printer:</label>
        <select 
          value={selectedPrinter} 
          onChange={(e) => setSelectedPrinter(e.target.value)}
        >
          {printers.map(printer => (
            <option key={printer.name} value={printer.name}>
              {printer.name} {printer.is_default ? '(Default)' : ''}
            </option>
          ))}
        </select>
      </div>

      <div>
        <input 
          type="file" 
          accept=".pdf,image/*" 
          onChange={handlePrintFile}
          disabled={loading}
        />
      </div>

      <div>
        <button onClick={handlePrintText} disabled={loading}>
          Print Text
        </button>
      </div>

      {status && <p>{status}</p>}
    </div>
  );
};

export default PrintComponent;
```

## Best Practices

1. **Environment Variables**: Store API tokens in environment variables
2. **Error Handling**: Always wrap API calls in try-catch blocks
3. **Rate Limiting**: Implement client-side rate limiting for bulk operations
4. **File Size**: Check file size before encoding to base64
5. **Timeouts**: Set appropriate timeouts for print operations
6. **Validation**: Validate content type and printer name before printing
